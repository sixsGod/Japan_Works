## 注释
```powershell
# 这是一个单行注释
Write-Output "Hello, PowerShell"

<#
这是多行注释,可用于文档说明
PowerShell 脚本文件使用 .ps1 后缀名,执行PowerShell用.\hello.ps1
#>
```

## 变量
```powershell
# 定义变量:变量以 $ 符号开头，无需事先声明类型
$name = "Alice"
$age = 25
# 使用变量
Write-Output "User: $($name), Age: $($age)"
```

## 数据类型
```powershell
$str = "Hello"    # 字符串
$pi = 3.14        # 小数
$isTrue = $true   # 布尔值
$arr = @(1, 2, 3) # 数组
$h = @{Name="Tom"; Age=30}  # 哈希表

# 可以使用 .GetType() 查看变量类型：
$str.GetType().Name   # String
```

## 运算符
```powershell
# 比较运算	 	-eq -ne -lt -gt
# 逻辑运算 	-and -or -not
# 字符串 -like -match -replace
# 包含运算 -in -contains
"abc" -like "a*"  # True
```

## 条件判断
```powershell
#  if-elseif-else
if ($score -ge 90) {
    "优秀"
} elseif ($score -ge 60) {
    "及格"
} else {
    "不及格"
}

# switch
switch ($value) {
    "start" { Write-Output "开始任务" }
    "stop"  { Write-Output "停止任务" }
    "exit"  { Write-Output "退出程序" }
    default { Write-Output "未知命令" }
}

```

## 循环结构
```powershell
# for 循环
for ($i = 1; $i -le 5; $i++) {
    Write-Output "第 $i 次循环"
}

# foreach 循环
$colors = @("Red", "Green", "Blue")
foreach ($color in $colors) {
    Write-Output "颜色：$color"
}
# or 管道版本
$colors | ForEach-Object { Write-Output "你好，$_" }

# while 循环
$count = 0
while ($count -lt 3) {
    Write-Output $count
    $count++
}

# do-while 与 do-until
$count = 0
do {
    Write-Output "当前值：$count"
    $count++
} while ($count -lt 3) # do 循环会 至少执行一次：

$count = 0
do {
    Write-Output "当前值：$count"
    $count++
} until ($count -ge 3) # do-until：直到条件为真才停止

```

## 函数定义
```powershell
function Say-Hello {
    param([string]$name)
    Write-Output "Hello, $name!"
}

Say-Hello -name "PowerShell"
# or 简洁写法
function Square($x) { return $x * $x }
Square 5  # 输出 25
```

## 错误处理
```powershell
try {
    Get-Item "C:\NotExist.txt"
} catch {  # [System.IO.FileNotFoundException]  / $_.Exception.Message
    Write-Output "找不到文件"
} finally {
    # 无论是否出错，都会执行（可选）
}
```
